SELECT category, COUNT (product_id) from products  group by category;

SELECT sum (quantity * price_per_unit) as total_revenue from order_items;

SELECT c.full_name, count (o.order_id) as order_quantity from customers c left join orders o on o.customer_id = c.customer_id
left join order_items oi on o.order_id = oi.order_id group by c.full_name;

select avg(average_cost) from (select sum (oi.quantity * oi.price_per_unit) as average_cost from orders o join order_items oi on oi.order_id = o.order_id group by o.order_id);

SELECT o.status, count(o.status) from orders o group by o.status;

SELECT p.category, count (p.product_name) from products p group by p.category HAVING count (p.product_name) >1;

select c.full_name, count(o.order_id) from customers c join orders o on c.customer_id = o.customer_id group by c.full_name having count (o.order_id) >1;

select p.product_name, sum (oi.quantity) from products p join order_items oi on p.product_id = oi.product_id group by p.product_id limit 1;
